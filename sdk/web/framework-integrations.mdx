```javascript Basic Setup
import { useEffect } from 'react';
import Linkrunner from 'linkrunner-web';

export default function App() {
  useEffect(() => {
    initLinkrunner();
  }, []);

  const initLinkrunner = async () => {
    await Linkrunner.configure({
      token: 'YOUR_PROJECT_TOKEN',
      debug: false // Optional: Enable debug logging (default: false)
    });
  };

  return (
    <div className="App">
      <header className="App-header">
        <h1>My React App</h1>
        <p>Linkrunner Web SDK is automatically tracking page views!</p>
      </header>
    </div>
  );
}
```

```typescript TypeScript Support
import { useEffect } from 'react';
import Linkrunner, { SDKConfig } from 'linkrunner-web';

export default function App() {
  useEffect(() => {
    initLinkrunner();
  }, []);

  const initLinkrunner = async () => {
    const config: SDKConfig = {
      token: 'YOUR_PROJECT_TOKEN',
      debug: true
    };

    await Linkrunner.configure(config);
  };

  return (
    <div className="App">
      <header className="App-header">
        <h1>My React App</h1>
        <p>Linkrunner Web SDK is automatically tracking page views!</p>
      </header>
    </div>
  );
}
```

```javascript pages/_app.js
import { useEffect } from 'react';
import Linkrunner from 'linkrunner-web';

export default function MyApp({ Component, pageProps }) {
  useEffect(() => {
    initLinkrunner();
  }, []);

  const initLinkrunner = async () => {
    await Linkrunner.configure({
      token: 'YOUR_PROJECT_TOKEN',
      debug: false // Optional: Enable debug logging (default: false)
    });
  };

  return (
    <>
      <Component {...pageProps} />
    </>
  );
}
```

```javascript app/layout.js (App Router)
'use client';
import { useEffect } from 'react';

      debug: false
    });
  };

  return (
    <html lang="en">
      <body>{children}</body>
    </html>
  );
}
```
