```javascript Method 2: Dynamic Loading with useEffect
import logo from './logo.svg';
import './App.css';
import React, { useEffect, useState } from 'react';

function App() {
  const [linkrunnerSDKLoaded, setLinkrunnerSDKLoaded] = useState(false);

  useEffect(() => {
    const script = document.createElement('script');
    script.src = 'https://sdk.linkrunner.io/latest/linkrunner-sdk.min.js';
    script.async = true;

    script.onload = () => {
      setLinkrunnerSDKLoaded(true);
      console.log('Linkrunner SDK loaded successfully');
    };

    document.body.appendChild(script);

    return () => {
      // Cleanup
      if (document.body.contains(script)) {
        document.body.removeChild(script);
      }
    };
  }, []);

  const handleInitViaDynamicLoading = async () => {
    if (window.LinkrunnerSDK) {
      const initOptions = {
        token: 'your-token',
        debug: true,
      };

      console.log('Initializing with options:', initOptions);
      const response = await window.LinkrunnerSDK.init(initOptions);
      alert(`Response: ${JSON.stringify(response)}`);
    } else {
      alert('Script not loaded yet!');
    }
  };

  return (
    <div className="App">
      <header className="App-header">
        <img src={logo} className="App-logo" alt="logo" />
        <p>
          Edit <code>src/App.js</code> and save to reload.
        </p>
        {linkrunnerSDKLoaded && (
          <button onClick={handleInitViaDynamicLoading} style={{
            padding: '10px 20px',
            fontSize: '16px',
            margin: '10px',
            cursor: 'pointer'
          }}>
            2. Test via dynamic loading
          </button>
        )}
        <a
          className="App-link"
          href="https://reactjs.org"
          target="_blank"
          rel="noopener noreferrer"
        >
          Learn React
        </a>
      </header>
    </div>
  );
}

export default App;
```
